const express = require('express'); const axios = require('axios'); const app = express(); const port = 3001; // Middleware app.use(express.json()); // In-memory order database let orders = {}; let orderIdCounter = 1; // Health check app.get('/health', (req, res) => { res.json({ status: 'Order Service is running!', timestamp: new Date().toISOString() }); }); // Get all orders app.get('/api/orders', (req, res) => { res.json({ success: true, data: Object.values(orders), count: Object.keys(orders).length }); }); // Get order by ID app.get('/api/orders/:id', (req, res) => { const id = parseInt(req.params.id); const order = orders[id]; if (order) { res.json({ success: true, data: order }); } else { res.status(404).json({ success: false, error: 'Order not found' }); } }); // Place a new order app.post('/api/orders', async (req, res) => { const { productId, quantity } = req.body; if (!productId || !quantity) { return res.status(400).json({ success: false, error: 'Missing fields', message: 'productId and quantity are required' }); } try { // Fetch product info from product-service const productRes = await axios.get(`http://product-service:3000/api/products/${productId}`); const product = productRes.data.data; // (Optional) Check stock if (product.stock < quantity) { return res.status(400).json({ success: false, error: 'Insufficient stock' }); } // Create order const newOrder = { id: orderIdCounter++, productId: product.id, productName: product.name, quantity, total: product.price * quantity, status: 'Pending', createdAt: new Date().toISOString() }; orders[newOrder.id] = newOrder; res.status(201).json({ success: true, data: newOrder, message: 'Order placed successfully' }); } catch (error) { return res.status(404).json({ success: false, error: 'Product not found or service unavailable', details: error.message }); } }); // Root endpoint app.get('/', (req, res) => { res.json({ message: 'Order Service API', version: '1.0.0', endpoints: { health: '/health', getAllOrders: '/api/orders', getOrderById: '/api/orders/:id', placeOrder: '/api/orders [POST]' } }); }); app.listen(port, '0.0.0.0', () => { console.log(`Order Service running on port ${port}`); });
